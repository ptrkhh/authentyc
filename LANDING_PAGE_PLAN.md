# Authentyc.AI Landing Page Plan
## Strategy A: Multi-Persona Testing Page

**Date**: 2025-11-21
**Goal**: Test demand across multiple categories (dating, hiring, founder matching), collect qualified waitlist signups, identify strongest segment for Phase 1 launch
**Timeline**: Build in 2-3 days, launch within 1 week
**Success Metric**: 100+ waitlist signups in Month 1, 300+ by Month 3, clear category winner (>40% of signups)

---

## Strategic Objectives

### Primary Goals
1. **Identify strongest category** across hiring, dating, and founder matching (>40% threshold)
2. **Validate demand** with qualified leads (email + use case + AI history)
3. **Test messaging** to see which value propositions resonate
4. **Build early adopter community** for validation phase

### Secondary Goals
5. **A/B test different positioning** (authenticity vs. capability vs. compatibility)
6. **Gauge willingness to pay** (via survey question, not blocking signup)
7. **Identify early adopters** for validation study recruitment
8. **Create shareable content** for Product Hunt, HN, Twitter

### Success Metrics (Conservative Approach)
- **Week 1**: 50 signups (mostly from personal network)
- **Month 1**: 100+ signups with 50%+ open rate on follow-up email
- **Month 3**: 300+ signups with clear category winner (>40% of signups)
- **Month 6**: 500+ active users in winning category
- **Conversion rate target**: 3-5% of visitors â†’ waitlist signup (conservative)

### API Route: `/api/waitlist/route.ts`

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@supabase/supabase-js';
import { z } from 'zod';

const waitlistSchema = z.object({
  email: z.string().email('Invalid email address'),
  primary_interest: z.enum([
    'hiring_recruiter',
    'hiring_jobseeker',
    'dating',
    'cofounder',
    'mastermind',
    'other'
  ]),
  other_interest_detail: z.string().optional(),
  has_ai_history: z.enum(['extensive', 'some', 'none']).optional(),
});

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const data = waitlistSchema.parse(body);

    const supabase = createClient(
      process.env.NEXT_PUBLIC_SUPABASE_URL!,
      process.env.SUPABASE_SERVICE_ROLE_KEY!
    );

    // Extract UTM params and metadata
    const url = new URL(request.url);
    const utm_source = url.searchParams.get('utm_source');
    const utm_medium = url.searchParams.get('utm_medium');
    const utm_campaign = url.searchParams.get('utm_campaign');
    const referrer = request.headers.get('referer');
    const user_agent = request.headers.get('user-agent');

    // Insert into database
    const { data: lead, error } = await supabase
      .from('waitlist_leads')
      .insert({
        ...data,
        source: utm_source || 'direct',
        utm_source,
        utm_medium,
        utm_campaign,
        referrer,
        user_agent,
      })
      .select()
      .single();

    if (error) {
      // Check for duplicate email
      if (error.code === '23505') {
        return NextResponse.json(
          { error: 'Email already on waitlist' },
          { status: 409 }
        );
      }
      throw error;
    }

    // Send welcome email via Resend
    if (process.env.RESEND_API_KEY) {
      await fetch('https://api.resend.com/emails', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${process.env.RESEND_API_KEY}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          from: 'Authentyc <hello@authentyc.ai>',
          to: data.email,
          subject: `Welcome to Authentyc [#${lead.id.slice(0, 8)}]`,
          html: getWelcomeEmailHTML(data.primary_interest),
        }),
      });
    }

    // Track in PostHog
    if (process.env.NEXT_PUBLIC_POSTHOG_KEY) {
      // Client-side PostHog will handle this
    }

    return NextResponse.json({
      success: true,
      message: 'Successfully joined waitlist',
      position: lead.id, // Could query count for actual position
    });

  } catch (error) {
    if (error instanceof z.ZodError) {
      return NextResponse.json(
        { error: 'Invalid form data', details: error.errors },
        { status: 400 }
      );
    }

    console.error('Waitlist error:', error);
    return NextResponse.json(
      { error: 'Failed to join waitlist' },
      { status: 500 }
    );
  }
}

function getWelcomeEmailHTML(interest: string): string {
  // Return HTML email template
  // TODO: Use React Email or MJML for better emails
  return `
    <h1>Welcome to Authentyc</h1>
    <p>You're officially on the waitlist for ${interest}.</p>
    <!-- Full email template here -->
  `;
}
```
